# Punctuations

# Use $...$ instead of \(\) in LaTeX
\\\((?!\\\))(.*?)\\\)
# $$$1$$

# Use \dots instead of ... in LaTeX.
\$([^$]*)\.\.\.([^$]*)\$
# $$$1$$

# scrunched up punctuation
([a-zA-Z"])([,.;:?!])([a-zA-Z])
# $1$2 $3

# Words and phrases

(M|m)ultilayered (P|p)erceptron
# $1ultilayer $2erceptron
